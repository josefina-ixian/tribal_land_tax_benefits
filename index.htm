    <script>

        document.addEventListener('DOMContentLoaded', () => {

           

            // --- Financial Stacking Chart ---

            const ctxStack = document.getElementById('benefitsChart').getContext('2d');

            new Chart(ctxStack, {

                type: 'bar',

                data: {

                    labels: ['Base (No PWA)', 'Base (w/ PWA)', 'PWA + Energy Comm.', 'Max Stacking'],

                    datasets: [

                        { label: 'Base ITC', data: [6, 30, 30, 30], backgroundColor: '#164e63' },

                        { label: 'Energy Comm.', data: [2, 0, 10, 10], backgroundColor: '#b45309' },

                        { label: 'Domestic Content', data: [0, 0, 0, 10], backgroundColor: '#065f46' }

                    ]

                },

                options: {

                    responsive: true,

                    maintainAspectRatio: false,

                    scales: { x: { stacked: true }, y: { stacked: true, max: 60, title: { display: true, text: 'Percentage (%)' } } },

                    plugins: { legend: { position: 'bottom' } }

                }

            });

 

            // --- Interactive Scenario Calculator ---

            const costSlider = document.getElementById('inputCost');

            const dispCost = document.getElementById('displayCost');

            const checkPWA = document.getElementById('checkPWA');

            const checkEC = document.getElementById('checkEC');

            const checkDC = document.getElementById('checkDC');

 

            const outRefund = document.getElementById('outRefund');

            const outRate = document.getElementById('outRate');

            const outNet = document.getElementById('outNet');

 

            const ctxSce = document.getElementById('scenarioChart').getContext('2d');

            let sceChart = new Chart(ctxSce, {

                type: 'doughnut',

                data: {

                    labels: ['Federal Grant', 'Tribe/Loan Share'],

                    datasets: [{ data: [40, 60], backgroundColor: ['#059669', '#1e293b'], borderWidth: 0 }]

                },

                options: { cutout: '75%', plugins: { legend: { position: 'bottom' } } }

            });

 

            const updateCalc = () => {

                const rawCost = parseInt(costSlider.value);

                const cost = rawCost * 1000000;

                dispCost.innerText = rawCost >= 1000 ? `$${(rawCost/1000).toFixed(1)} Billion` : `$${rawCost} Million`;

 

                let rate = checkPWA.checked ? 30 : 6;

                if (checkEC.checked) rate += (checkPWA.checked ? 10 : 2);

                if (checkDC.checked) rate += (checkPWA.checked ? 10 : 2);

 

                const refund = cost * (rate / 100);

                const net = cost - refund;

 

                outRefund.innerText = `$${refund.toLocaleString()}`;

                outRate.innerText = `${rate}%`;

                outNet.innerText = `$${net.toLocaleString()}`;

 

                sceChart.data.datasets[0].data = [rate, 100 - rate];

                sceChart.update();

            };

 

            [costSlider, checkPWA, checkEC, checkDC].forEach(el => el.addEventListener('input', updateCalc));

            updateCalc();

        });

    </script>
